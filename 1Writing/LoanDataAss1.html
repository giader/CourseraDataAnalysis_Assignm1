loansdata$emplength <- as.numeric(sub("10+", "11", emplength, fixed = TRUE))
rm(emplength)

## and the Monthly Income fundedPerc as the ratio between the Amount Funded

loansdata$fundedPerc <- loansdata$funded/loansdata$requested
